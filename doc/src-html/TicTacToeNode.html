001import java.util.*;
002/**
003 * TicTacToe runs a game of TicTacToe.
004 * 
005 * The class TicTacToe contains game logic needed for
006 * variable sized game of tic-tac-toe.
007 * 
008 * @author Arttu Knuutinen
009 * @version 2018.1214
010 */
011
012public class TicTacToe {
013    /**
014     * Scanner to take user input.
015     */
016    private static Scanner sc = new Scanner(System.in);
017    /**
018     * Minimum board size.
019     */
020    private static int minBoardSize = 3;
021    /**
022     * Maximum reasonable board size.
023     */
024    private static int maxReasonableSize = 25;
025    /**
026     * Board height, chosen by user.
027     */
028    private static int boardHeight = 0;
029    /**
030     * Board width, chosen by user.
031     */
032    private static int boardWidth = 0;
033    /**
034     * Marks needed in a row to win the game, chosen by user.
035     */
036    private static int inRowToWin = 0;
037    /**
038     * Game board in 2-dimensional array.
039     */
040    private static TicTacToeNode[][] board;
041    /**
042     * Boolean that keeps track of gamestate.
043     */
044    private static boolean gameOver = false;
045    /**
046     * Boolean that decides if it is the player's or the computer's turn.
047     */
048    private static boolean playerTurn = true;
049    /**
050     * Counts the turns taken, used to check if game ends in a draw.
051     */
052    private static int turnsTaken = 0;
053    /**
054     * Keeps track of the most recently played position.
055     */
056    private static TicTacToeNode lastUsedNode;
057
058    /**
059     * The Main method, runs at program start.
060     * 
061     * @param args command line arguments, not used.
062     */
063    public static void main(String[] args) {
064        startGame();
065        takeTurns();
066        endGame();
067    }
068    /**
069     * Runs necessary tasks for starting the game.
070     */
071    private static void startGame(){
072        String intro = "You're playing TicTacToe against the computer.\n" +
073                        "You're using \"X\"\n" +
074                        "Computer uses \"O\"\n" +
075                        "Please select the size of the board" +
076                        " and amount of marks needed to win.";
077        System.out.println(intro);
078        createBoard();
079        printBoard();
080    }
081    /**
082     * Creates the board with user input.
083     */
084    private static void createBoard(){
085        while(boardHeight < minBoardSize){
086            System.out.print("Enter board height: ");
087            boardHeight = getUserInputInt(sc.nextLine());
088            if(boardHeight > maxReasonableSize){
089                System.out.println("That's pretty big!");
090                if(!getUserConfirmation()){
091                    boardHeight = 0;
092                }
093            }
094        }
095        while(boardWidth < minBoardSize){
096            System.out.print("Enter board width: ");
097            boardWidth = getUserInputInt(sc.nextLine());
098            if(boardWidth > maxReasonableSize){
099                System.out.println("That's pretty big!");
100                if(!getUserConfirmation()){
101                    boardWidth = 0;
102                }
103            }
104        }
105        int minMax = Math.min(boardWidth, boardHeight);
106        int minWinCond = minMax >= 10 ? 5 : 3;
107        while(inRowToWin < minBoardSize || inRowToWin > minMax) {
108            System.out.print("Enter amount of marks needed in a row to win"
109                            + "(" + minWinCond + "-" + minMax + "): ");
110            inRowToWin = getUserInputInt(sc.nextLine());
111        }
112        board = new TicTacToeNode[boardHeight][boardWidth];
113        for (int i = 0; i < board.length; i++) {
114            for (int j = 0; j < board[i].length; j++) {
115                board[i][j] = new TicTacToeNode(j, i);
116            }
117        }
118    }
119    /**
120     * Alternates turns between the player and the computer.
121     */
122    private static void takeTurns(){
123        while(!gameOver){
124            if(playerTurn){
125                playerTurn();
126            } else {
127                aiTurn();
128            }
129            printBoard();
130            gameOver = wasWinningMove(lastUsedNode);
131            if(!gameOver){
132                turnsTaken++;
133                if(turnsTaken == boardWidth * boardHeight){
134                    gameOver = true;
135                }
136                playerTurn = !playerTurn;
137            }
138        }
139    }
140    /**
141     * Handles the player's turn.
142     */
143    private static void playerTurn(){
144        String input = "";
145        int x = -1;
146        int y = -1;
147        boolean turnComplete = false;
148        String prompt = "Select coordinates \"x, y\" where to play your mark";
149        String coordinateError = "Please enter coordinates as requested.";
150        while(!turnComplete) {
151            while(x < 0 || x >= boardWidth || y < 0 || y >= boardHeight){
152                System.out.println(prompt);
153                System.out.print("Enter coordinates: ");
154                input = sc.nextLine();
155                input = input.replaceAll(" ", "");
156                String[] temp = input.split("\\D");
157                if(temp.length == 2){
158                    if(temp[0].length() > 0){
159                        x = Integer.parseInt(temp[0]) - 1;
160                    }
161                    if(temp[1].length() > 0){
162                        y = Integer.parseInt(temp[1]) - 1;
163                    }
164                } else {
165                    System.out.println(coordinateError);
166                }
167            }
168            if(!board[y][x].isUsed()){
169                board[y][x].playMark('X');
170                turnComplete = true;
171            } else {
172                System.out.println("You cannot place a mark over another one");
173                x = -1;
174                y = -1;
175            }
176        }
177        lastUsedNode = board[y][x];
178    }
179    /**
180     * Handles the computer's turn.
181     * 
182     * Tries to play its move somewhere around the player's last position.
183     * If there are no empty nodes around the position, plays on a random node.
184     */
185    private static void aiTurn(){
186        int y = 0;
187        int x = 0;
188        int playerLastX = lastUsedNode.getX();
189        int playerLastY = lastUsedNode.getY();
190
191        // Finds empty nodes around the last played node
192        ArrayList<TicTacToeNode> freeNodes = new ArrayList<TicTacToeNode>();
193        for (int i = -1; i < 2; i++) {
194            for (int j = -1; j < 2; j++) {
195                int tempX = playerLastX + j;
196                int tempY = playerLastY + i;
197                if(isBetween(tempX, 0, boardWidth -1)
198                        && isBetween(tempY, 0, boardHeight -1)) {
199                    if(!board[tempY][tempX].isUsed()){
200                        freeNodes.add(board[tempY][tempX]);
201                    }
202                }
203            }
204        }
205        // Plays on a random free node if any are found
206        if(freeNodes.size() > 0){
207            int i = (int)(Math.random() * freeNodes.size());
208            x = freeNodes.get(i).getX(); 
209            y = freeNodes.get(i).getY(); 
210        } else {
211            do {
212                y = (int)(Math.random() * boardHeight);
213                x = (int)(Math.random() * boardWidth);    
214            } while (board[y][x].isUsed());        
215        }
216        board[y][x].playMark('O');
217        lastUsedNode = board[y][x];
218        System.out.println("Computer played on \"x:" + (x +1)
219                            + ", y:" + (y +1) + "\"");
220    }
221    /**
222     * Reports the game result.
223     */
224    private static void endGame(){
225        String endText = "";
226        if(turnsTaken == boardWidth * boardHeight){
227            endText = "It's a draw.";
228        } else {
229            endText = (playerTurn ? "You win " : "Computer wins " )
230                    + "from position \"x:" + (lastUsedNode.getX() + 1)
231                    + ", y:" + (lastUsedNode.getY() + 1) + "\"!";
232        }
233        System.out.println(endText);
234    }
235    /**
236     * Returns parsed integer from user input string.
237     * 
238     * @param input String from user.
239     * @return -1 if user didn't give valid input.
240     */
241    private static int getUserInputInt(String input) {
242        int output = -1;
243        try {
244            output = Integer.parseInt(input);
245        } catch (NumberFormatException e) {
246            System.out.println("Enter a valid number");
247        }
248        return output;
249    }
250    /**
251     * Prints board coordinate numbers for X - axis.
252     */
253    private static void printXCoordinates(){
254        String line = "   X";
255        for (int i = 1; i <= boardWidth; i++) {
256            line += String.format("% 3d", i) + " ";
257        }
258        System.out.println(line);
259    }
260    /**
261     * Prints a line of ascii graphic for the board.
262     * 
263     * @param printCoordinate determines if to print Y - axis identifier.
264     */
265    private static void printAsciiLine(boolean printCoordinate){
266        String line = printCoordinate ? " Y " : "   ";
267        for (int i = 0; i < boardWidth; i++) {
268            line += "+---";
269        }
270        System.out.println(line + "+");
271    }
272    /**
273     * Prints marks from the board from given index.
274     * 
275     * @param i the board index to print.
276     */
277    private static void printBoardMarks(int i){
278        for (int j = 0; j < board[i].length; j++) {
279            if(j == 0){
280                System.out.print(String.format("% 3d", i + 1)
281                                + "| " + board[i][j].getMark());
282            } else if (j == boardWidth - 1){
283                System.out.print(" | " + board[i][j].getMark() + " |");
284            } else {
285                System.out.print(" | " + board[i][j].getMark());
286            }
287        }
288        System.out.println();
289    }
290    /**
291     * Prints the game board.
292     */
293    private static void printBoard(){
294        printXCoordinates();
295        for (int i = 0; i < board.length; i++) {
296            printAsciiLine(i == 0);
297            printBoardMarks(i);
298        }
299        printAsciiLine(false);
300        System.out.println();
301    }
302    /**
303     * Determines if the node forms a winning combination.
304     * 
305     * @param node the node to check for victory.
306     * @return true if node forms a combination of required length.
307     */
308    private static boolean wasWinningMove(TicTacToeNode node){
309        char c = node.getMark();
310        int inRowCount = 0;
311        boolean victory = false;
312        int xPos = node.getX();
313        int yPos = node.getY();
314        
315        // Horizontal check
316        for (int i = -(inRowToWin -1); i <= (inRowToWin -1) && !victory; i++) {
317            if(isBetween(xPos + i, 0, boardWidth -1)) {
318                if(board[yPos][xPos + i].getMark() == c){
319                    inRowCount++;
320                    if(inRowCount == inRowToWin){
321                        victory = true;
322                        break;
323                    }
324                } else {
325                    inRowCount = 0;
326                }    
327            }
328        }
329
330        // Vertical check
331        inRowCount = 0;
332        for (int i = -(inRowToWin -1); i <= (inRowToWin -1) && !victory; i++) {
333            if(isBetween(yPos + i, 0, boardHeight -1)) {
334                if(board[yPos + i][xPos].getMark() == c){
335                    inRowCount++;
336                    if(inRowCount == inRowToWin){
337                        victory = true;
338                        break;
339                    }
340                } else {
341                    inRowCount = 0;
342                }    
343            }
344        }
345
346        // Diagonals
347
348        // North-west to south-east
349        inRowCount = 0;
350        for (int i = -(inRowToWin -1); i <= (inRowToWin -1) && !victory; i++) {
351            if(isBetween(xPos + i, 0, boardWidth -1) 
352                    && isBetween(yPos + i, 0, boardHeight -1)) {
353                if(board[yPos + i][xPos + i].getMark() == c){
354                    inRowCount++;
355                    if(inRowCount == inRowToWin){
356                        victory = true;
357                        break;
358                    }
359                } else {
360                    inRowCount = 0;
361                }
362            }
363        }
364
365        // South-west to north-east
366        inRowCount = 0;
367        for (int i = -(inRowToWin -1); i <= (inRowToWin -1) && !victory; i++) {
368            if(isBetween(xPos + i, 0, boardWidth -1) 
369                    && isBetween(yPos - i, 0, boardHeight -1)) {
370                if(board[yPos - i][xPos + i].getMark() == c){
371                    inRowCount++;
372                    if(inRowCount == inRowToWin){
373                        victory = true;
374                        break;
375                    }
376                } else {
377                    inRowCount = 0;
378                }
379            }
380        }
381        return victory;
382    }
383    /**
384     * Checks if x is between min and max.
385     * 
386     * @param x value to test.
387     * @param min the low value.
388     * @param max the high value.
389     * 
390     * @return true if value of x is between min and max
391     */
392    private static boolean isBetween(int x, int min, int max){
393        return (x >= min && x <= max);
394    }
395    /**
396     * Asks user to confirm a decision.
397     * 
398     * @return true if user input starts with 'y'
399     */
400    private static boolean getUserConfirmation(){
401        String input = "";
402        boolean success = false;
403        char c = ' ';
404        do {
405            try {
406                System.out.println("Are you sure you want to do that? (y/n)");
407                input = sc.nextLine().toLowerCase();
408                c = input.charAt(0);
409                if(c == 'y' || c == 'n'){
410                    success = true;
411                }
412            } catch (Exception e) {
413
414            }
415        } while (!success);
416        return c == 'y';
417    }

420/**
421 * TicTacToeNode stores useful data for TicTacToe game-logic.
422 * 
423 * @author Arttu Knuutinen
424 * @version 2018.1214
425 */
426class TicTacToeNode {
427    /**
428     * Boolean used to see if a mark has already been placed on the node.
429     */
430    private boolean isUsed = false;
431    /**
432     * Stores the mark that's placed on the node.
433     */
434    private char mark = ' ';
435    /**
436     * The node's X - index on the board.
437     */
438    private int xPos;
439    /**
440     * The node's Y - index on the board.
441     */
442    private int yPos;
443
444    /**
445     * Get method for the X - index.
446     * 
447     * @return xPos
448     */
449    public int getX(){
450        return xPos;
451    }
452    /**
453     * Get method for the Y - index.
454     * 
455     * @return yPos
456     */
457    public int getY(){
458        return yPos;
459    }
460    /**
461     * Get method for the stored mark.
462     * 
463     * @return the "mark" char.
464     */
465    public char getMark(){
466        return mark;
467    }
468    /**
469     * Retuns the value of isUsed boolean.
470     * 
471     * @return true if a mark has been played on the node.
472     */
473    public boolean isUsed(){
474        return isUsed;
475    }
476    /**
477     * Sets the mark variable to be the given parameter.
478     * 
479     * @param mark char to set on the node.
480     */
481    private void setMark(char mark){
482        this.mark = mark;
483    }
484    /**
485     * Used to play a mark on the node, sets isUsed to true.
486     * 
487     * @param mark the char to play on the node.
488     */
489    public void playMark(char mark){
490        setMark(mark);
491        isUsed = true;
492    }
493    /**
494     * Constructor
495     * 
496     * @param xPos the X - index of the node in 2D-array
497     * @param yPos the Y - index of the node in 2D-array
498     */
499    public TicTacToeNode(int xPos, int yPos){
500        this.xPos = xPos;
501        this.yPos = yPos;
502    }
503}
